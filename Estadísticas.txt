function normalizePromo(v) {

}

              <div style="font-family:var(--mono);font-size:0.55rem;letter-spacing:0.15em;color:var(--text-mid);margin-bottom:6px">PAGADOS</div>
              <div style="font-family:var(--serif);font-size:2.4rem;font-weight:700;color:var(--green)">${pagados.length}</div>
              <div style="font-size:0.8rem;color:var(--text-mid);font-style:italic">S/ ${ingresos.toLocaleString('es-PE')}</div>
            </div>
            <div style="text-align:center;padding:16px;background:var(--crema);border:1px solid var(--crema3)">
              <div style="font-family:var(--mono);font-size:0.55rem;letter-spacing:0.15em;color:var(--text-mid);margin-bottom:6px">CORTESÍA</div>
              <div style="font-family:var(--serif);font-size:2.4rem;font-weight:700;color:var(--text-mid)">${gratis.length}</div>
              <div style="font-size:0.8rem;color:var(--text-mid);font-style:italic">S/ 0</div>
            </div>
          </div>
          <div class="promo-bar-wrap"><div class="promo-bar-fill" id="promo-fill" style="width:0%"></div></div>
          <div style="display:flex;justify-content:space-between;font-family:var(--mono);font-size:0.58rem;letter-spacing:0.08em;color:var(--text-mid)">
            <span>${pct(pagados.length,total)}% pagados</span>
            <span>${pct(gratis.length,total)}% cortesía</span>
          </div>
        </div>
      </div>
    </div>`;

  setTimeout(() => { const f=document.getElementById('promo-fill'); if(f) f.style.width=pct(pagados.length,total)+'%'; }, 100);

  // Store normalized for table
  window._normalized = normalized;
  filteredBoletos = [...normalized];
  applyFilters();
  buildCharts(pagados.length, gratis.length, areas);
}

function applyFilters() {
  const q     = (document.getElementById('search-input')?.value||'').toLowerCase();
  const promo = document.getElementById('filter-promo')?.value||'';
  const area  = document.getElementById('filter-area')?.value||'';
  const src   = window._normalized || [];

  filteredBoletos = src.filter(b => {
    const matchQ = !q || b.nombre.toLowerCase().includes(q) || b.dni.includes(q) || b.id.toString().includes(q) || b.area.toLowerCase().includes(q);
    const matchP = !promo || b.promo === promo;
    const matchA = !area || b.area === area;
    return matchQ && matchP && matchA;
  });
  currentPage = 1;
  renderTable();
}

function renderTable() {
  const tbody = document.getElementById('tickets-tbody');
  if (!tbody) return;
  const start = (currentPage-1)*PAGE_SIZE;
  const slice = filteredBoletos.slice(start, start+PAGE_SIZE);
  const total = filteredBoletos.length;
  const pages = Math.ceil(total/PAGE_SIZE);

  tbody.innerHTML = slice.map(b => {
    const pagado = b.promo === 'No';
    return `<tr>
      <td><span class="ticket-id">${b.id||'—'}</span></td>
      <td style="font-style:italic">${b.nombre||'—'}</td>
      <td style="font-family:var(--mono);font-size:0.75rem">${b.dni||'—'}</td>
      <td style="font-family:var(--mono);font-size:0.75rem">${b.tel||'—'}</td>
      <td><span style="font-family:var(--mono);font-size:0.7rem">${b.area}</span></td>
      <td><span class="badge ${pagado?'pagado':'gratis'}">${pagado?'PAGADO':'CORTESÍA'}</span></td>
      <td style="font-family:var(--serif);font-size:1rem;font-weight:600;color:${pagado?'var(--green)':'var(--text-mid)'}">${pagado?'S/ 5.00':'—'}</td>
    </tr>`;
  }).join('') || `<tr><td colspan="7" style="text-align:center;padding:28px;color:var(--text-mid);font-style:italic">Sin resultados</td></tr>`;

  document.getElementById('page-info').textContent = `${start+1}–${Math.min(start+PAGE_SIZE,total)} de ${total}`;
  document.getElementById('btn-prev').disabled = currentPage<=1;
  document.getElementById('btn-next').disabled = currentPage>=pages;
}

function changePage(dir) {
  currentPage = Math.max(1, Math.min(currentPage+dir, Math.ceil(filteredBoletos.length/PAGE_SIZE)));
  renderTable();
  document.getElementById('tickets-section')?.scrollIntoView({behavior:'smooth',block:'start'});
}

function sortBy(col) {
  if(sortCol===col) sortDir*=-1; else { sortCol=col; sortDir=1; }
  filteredBoletos.sort((a,b) => (a[col]||'').toString().localeCompare((b[col]||'').toString(),'es')*sortDir);
  currentPage=1; renderTable();
}

function buildCharts(nP, nG, areas) {
  Chart.defaults.font.family = "'DM Mono', monospace";
  Chart.defaults.color = '#6b4a4a';
  if(chartPromo) chartPromo.destroy();
  chartPromo = new Chart(document.getElementById('chart-promo'), {
    type:'doughnut',
    data:{ labels:['Pagados (S/5)','Cortesía'],
      datasets:[{ data:[nP,nG], backgroundColor:['#3a7d5a','#dfd6c2'], borderColor:['#fff','#fff'], borderWidth:3, hoverOffset:10 }]
    },
    options:{ responsive:true, maintainAspectRatio:false, cutout:'65%',
      plugins:{ legend:{ labels:{ font:{size:11}, padding:20 } },
        tooltip:{ callbacks:{ label:c=>` ${c.label}: ${c.parsed} (${pct(c.parsed,nP+nG)}%)` } } }
    }
  });
  if(chartAreas) chartAreas.destroy();
  const top = areas.slice(0,8);
  chartAreas = new Chart(document.getElementById('chart-areas'), {
    type:'bar',
    data:{ labels:top.map(a=>a.name.length>13?a.name.slice(0,13)+'…':a.name),
      datasets:[{ label:'S/ Ingresos', data:top.map(a=>a.ingresos),
        backgroundColor:top.map((_,i)=>i===0?'#b8943a':'#7a1420'), borderRadius:3 }]
    },
    options:{ responsive:true, maintainAspectRatio:false,
      plugins:{ legend:{display:false}, tooltip:{callbacks:{label:c=>` S/ ${c.parsed.y.toLocaleString('es-PE')}`}} },
      scales:{ x:{ticks:{font:{size:9}},grid:{color:'#f0ece8'}}, y:{ticks:{font:{size:9},callback:v=>'S/'+v},grid:{color:'#f0ece8'}} }
    }
  });
}
